364template<typename T>
365inline CountingBloomFilter<T>::CountingBloomFilter(size_t bytes,
366 unsigned hash_num,
367 std::string hash_fn)
368 : bytes(
369 size_t(std::ceil(double(bytes) / sizeof(uint64_t)) * sizeof(uint64_t)))
370 , array_size(get_bytes() / sizeof(array[0]))
371 , hash_num(hash_num)
372 , hash_fn(std::move(hash_fn))
373 , array(new std::atomic<T>[array_size])
374{
375 check_error(bytes == 0, "CountingBloomFilter: memory budget must be >0!");
376 check_error(hash_num == 0,
377 "CountingBloomFilter: number of hash values must be >0!");
378 check_error(
379 hash_num > MAX_HASH_VALUES,
380 "CountingBloomFilter: number of hash values cannot be over 1024!");
381 check_warning(sizeof(uint8_t) != sizeof(std::atomic<uint8_t>),
382 "Atomic primitives take extra memory. CountingBloomFilter will "
383 "have less than " +
384 std::to_string(bytes) + " for bit array.");
385 std::memset((void*)array.get(), 0, array_size * sizeof(array[0]));
386}

388template<typename T>
389inline void
390CountingBloomFilter<T>::insert(const uint64_t* hashes)
391{
392 // Update flag to track if increment is done on at least one counter
393 bool update_done = false;
394 T new_val;
395 T min_val = contains(hashes);
396 while (!update_done) {
397 // Simple check to deal with overflow
398 new_val = min_val + 1;
399 if (min_val > new_val) {
400 return;
401 }
402 for (size_t i = 0; i < hash_num; ++i) {
403 decltype(min_val) temp_min_val = min_val;
404 if (array[hashes[i] % array_size].compare_exchange_strong(temp_min_val,
405 new_val)) {
406 update_done = true;
407 }
408 }
409 // Recalculate minval because if increment fails, it needs a new minval to
410 // use and if it doesnt hava a new one, the while loop runs forever.
411 if (!update_done) {
412 min_val = contains(hashes);
413 }
414 }
415}

418inline T
419CountingBloomFilter<T>::contains(const uint64_t* hashes) const
420{
421 T min = array[hashes[0] % array_size];
422 for (size_t i = 1; i < hash_num; ++i) {
423 const size_t idx = hashes[i] % array_size;
424 if (array[idx] < min) {
425 min = array[idx];
426 }
427 }
428 return min;

505{
506 /* Initialize cpptoml root table
507 Note: Tables and fields are unordered
508 Ordering of table is maintained by directing the table
509 to the output stream immediately after completion */
510 auto root = cpptoml::make_table();
511
512 /* Initialize bloom filter section and insert fields
513 and output to ostream */
514 auto header = cpptoml::make_table();
515 header->insert("bytes", get_bytes());
516 header->insert("hash_num", get_hash_num());
517 if (!hash_fn.empty()) {
518 header->insert("hash_fn", hash_fn);

521 root->insert(COUNTING_BLOOM_FILTER_MAGIC_HEADER, header);
