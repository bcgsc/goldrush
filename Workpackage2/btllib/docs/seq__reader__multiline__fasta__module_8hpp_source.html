34inline bool
35SeqReaderMultilineFastaModule::buffer_valid(const char* buffer,
36 const size_t size)
37{
38 size_t current = 0;
39 unsigned char c;
40 enum State
41 {
42 IN_HEADER_1,
43 IN_HEADER_2,
44 IN_SEQ,
45 IN_TRANSITION
46 };
47 State state = IN_HEADER_1;
48 while (current < size) {
49 c = buffer[current];
50 switch (state) {
51 case IN_HEADER_1:
52 if (c == '>') {
53 state = IN_HEADER_2;
54 } else {
55 return false;
56 }
57 break;
58 case IN_HEADER_2:
59 if (c == '\n') {
60 state = IN_SEQ;
61 }
62 break;
63 case IN_SEQ:
64 if (c == '\n') {
65 state = IN_TRANSITION;
66 } else if (c != '\r' && !bool(COMPLEMENTS[c])) {
67 return false;
68 }
69 break;
70 case IN_TRANSITION:
71 if (c == '>') {
72 state = IN_HEADER_2;
73 break;
74 } else if (c != '\r' && !bool(COMPLEMENTS[c])) {
75 return false;
76 }
77 state = IN_SEQ;
78 break;
79 }
80 current++;
81 }
82 return true;
83}

86inline bool
87SeqReaderMultilineFastaModule::read_buffer(ReaderType& reader,
88 RecordType& record)
89{
90 record.header.clear();
91 record.seq.clear();
92 record.qual.clear();
93 if (reader.buffer.start < reader.buffer.end) {
94 int c;
95 for (;;) {
96 switch (stage) {
97 case Stage::HEADER: {
98 if (!reader.readline_buffer_append(record.header)) {
99 return false;
100 }
101 stage = Stage::SEQ;
102 }
103 // fall through
104 case Stage::SEQ: {
105 if (!reader.readline_buffer_append(record.seq)) {
106 return false;
107 }
108 rtrim(record.seq);
109 stage = Stage::TRANSITION;
110 }
111 // fall through
112 case Stage::TRANSITION: {
113 c = reader.getc_buffer();
114 if (c == EOF) {
115 return false;
116 }
117 reader.ungetc_buffer(c);
118 if (c == '>') {
119 stage = Stage::HEADER;
120 return true;
121 }
122 stage = Stage::SEQ;
123 break;
124 }
125 default: {
126 log_error("SeqReader has entered an invalid state.");
127 std::exit(EXIT_FAILURE); // NOLINT(concurrency-mt-unsafe)
128 }
129 }
130 }
131 }
132 return false;
133}
