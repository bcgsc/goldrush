89 std::exit(EXIT_FAILURE); // NOLINT(concurrency-mt-unsafe)

93 std::string output;
94 output.reserve(1 + id.size() + 1 + comment.size() + 1 + seq.size() + 3 +
95 qual.size() + 1);
96 output += headerchar;
97 if (!id.empty()) {
98 output += id;
99 }
100 if (!comment.empty()) {
101 output += " ";
102 output += comment;
103 }
104 output += '\n';
105
106 output += seq;
107 output += '\n';
108
109 if (format == FASTQ) {
110 check_error(seq.size() != qual.size(),
111 "Quality must be the same length as sequence.");
112 output += "+\n";
113 output += qual;
114 output += '\n';
115 }
116
117 {
118 std::unique_lock<std::mutex> lock(mutex);
119 check_error(fwrite(output.c_str(), 1, output.size(), sink) != output.size(),
120 "SeqWriter: fwrite failed.");
121 }
122}
