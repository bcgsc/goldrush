37pop_cnt_byte(uint8_t x)
38{
39 return ((0x876543210 >> // NOLINT

42 0xf; // NOLINT

618 unsigned hash_num,

631 "BloomFilter: number of hash values cannot be over 1024!");
632 check_warning(
633 sizeof(uint8_t) != sizeof(std::atomic<uint8_t>),
634 "Atomic primitives take extra memory. BloomFilter will have less than " +
635 std::to_string(bytes) + " for bit array.");
636 std::memset((void*)array.get(), 0, array_size * sizeof(array[0]));
637}

639inline void
640BloomFilter::insert(const uint64_t* hashes)
641{
642 for (unsigned i = 0; i < hash_num; ++i) {
643 const auto normalized = hashes[i] % array_bits;
644 array[normalized / CHAR_BIT] |= BIT_MASKS[normalized % CHAR_BIT];
645 }
646}

648inline bool
649BloomFilter::contains(const uint64_t* hashes) const
650{
651 for (unsigned i = 0; i < hash_num; ++i) {
652 const auto normalized = hashes[i] % array_bits;
653 const auto mask = BIT_MASKS[normalized % CHAR_BIT];
654 if (!bool(array[normalized / CHAR_BIT] & mask)) {
655 return false;
656 }
657 }
658 return true;
659}
660
661inline bool
662BloomFilter::contains_insert(const uint64_t* hashes)
663{
664 uint8_t found = 1;
665 for (unsigned i = 0; i < hash_num; ++i) {
666 const auto normalized = hashes[i] % array_bits;
667 const auto bitpos = normalized % CHAR_BIT;
668 const auto mask = BIT_MASKS[bitpos];
669 found &= ((array[normalized / CHAR_BIT].fetch_or(mask) >> bitpos) & 1);
670 }
671 return bool(found);
672}

674inline uint64_t
675BloomFilter::get_pop_cnt() const
676{
677 uint64_t pop_cnt = 0;
678#pragma omp parallel for default(none) reduction(+ : pop_cnt)
679 for (size_t i = 0; i < array_size; ++i) {
680 pop_cnt += pop_cnt_byte(array[i]);
681 }
682 return pop_cnt;
683}

687{

692BloomFilter::get_fpr() const

697inline std::shared_ptr<cpptoml::table>
698BloomFilterInitializer::parse_header(std::ifstream& file,
699 const std::string& magic_string)
700{
701 const std::string magic_with_brackets = std::string("[") + magic_string + "]";
702
703 std::string line;
704 std::getline(file, line);
705 if (line != magic_with_brackets) {
706 log_error(
707 std::string("Magic string does not match (likely version mismatch)\n") +
708 "File magic string:\t" + line + "\n" + "Loader magic string:\t" +
709 magic_with_brackets);
710 std::exit(EXIT_FAILURE); // NOLINT(concurrency-mt-unsafe)
711 }
712
713 /* Read bloom filter line by line until it sees "[HeaderEnd]"
714 which is used to mark the end of the header section and
715 assigns the header to a char array*/
716 std::string toml_buffer(line + '\n');
717 bool header_end_found = false;
718 while (bool(std::getline(file, line))) {
719 toml_buffer.append(line + '\n');
720 if (line == "[HeaderEnd]") {
721 header_end_found = true;
722 break;
723 }
724 }
725 if (!header_end_found) {
726 log_error("Pre-built bloom filter does not have the correct header end.");
727 std::exit(EXIT_FAILURE); // NOLINT(concurrency-mt-unsafe)
728 }
729 for (unsigned i = 0; i < PLACEHOLDER_NEWLINES; i++) {
730 std::getline(file, line);
731 }
732
733 // Send the char array to a stringstream for the cpptoml parser to parse
734 std::istringstream toml_stream(toml_buffer);
735 cpptoml::parser toml_parser(toml_stream);
736 const auto header_config = toml_parser.parse();
737
738 // Obtain header values from toml parser and assign them to class members
739 return header_config->get_table(magic_string);
740}

786BloomFilter::save(const std::string& path)
787{
788 /* Initialize cpptoml root table
789 Note: Tables and fields are unordered
