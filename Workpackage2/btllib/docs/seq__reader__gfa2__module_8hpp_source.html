37inline bool
38SeqReaderGfa2Module::buffer_valid(const char* buffer, const size_t size)
39{
40 const unsigned char specs[] = { 'H', 'S', 'F', 'E', 'G', 'O', 'U' };
41
42 enum State
43 {
44 IN_ID,
45 IN_ID_TAB,
46 IN_REST,
47 IN_IGNORED
48 };
49
50 auto is_a_spec = [&](unsigned char c) {
51 bool found = false;
52 for (unsigned char spec : specs) {
53 if (c == spec) {
54 found = true;
55 break;
56 }
57 }
58 return found;
59 };
60
61 State state = is_a_spec(buffer[0]) ? IN_ID : IN_IGNORED;
62 bool has_id = false;
63 size_t current = 0;
64 unsigned char c;
65 while (current < size) {
66 c = buffer[current];
67 switch (state) {
68 case IN_ID:
69 if (!is_a_spec(c)) {
70 return false;
71 }
72 has_id = true;
73 state = IN_ID_TAB;
74 break;
75 case IN_ID_TAB:
76 if (c != '\t') {
77 return false;
78 }
79 state = IN_REST;
80 break;
81 case IN_REST:
82 if (c == '\n') {
83 if (current + 1 < size) {
84 state = is_a_spec(buffer[current + 1]) ? IN_ID : IN_IGNORED;
85 }
86 }
87 break;
88 case IN_IGNORED:
89 if (c == '\n') {
90 if (current + 1 < size) {
91 state = is_a_spec(buffer[current + 1]) ? IN_ID : IN_IGNORED;
92 }
93 }
94 break;
95 default:
96 break;
97 }
98 current++;
99 }
100
101 return has_id;
102}
